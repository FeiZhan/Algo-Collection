#include <fstream>

#include <iostream>
#include <stdio.h>
using namespace std;

#include <memory.h>
#define MAX_NUM 110
int num, ans[MAX_NUM];
bool if_find;
void dfs(int digit_sum, int rest)
{
	if (digit_sum > 100)
		return;
	rest %= num;
	if (rest == 0)
	{
		if_find = true;
		for (int i = 0; i < digit_sum; i ++)
		{
			cout << ans[i];
		}
		cout << endl;
		return;
	}
	ans[digit_sum] = 0;
	dfs(digit_sum + 1, rest * 10);
	if (if_find == true)
		return;
	ans[digit_sum] = 1;
	dfs(digit_sum + 1, rest * 10 + 1);
	if (if_find == true)
		return;
}

int main()
{
#ifdef _FILE_DEBUG_
	ifstream fin;
	fin.open("../input.txt");
	cin.rdbuf(fin.rdbuf()); // assign file's streambuf to cin
#ifdef _C_LAN_
	freopen("../input.txt", "r", stdin);
#endif
#endif
#ifdef _FILE_DEBUG_
	ofstream fout;
	fout.open("../output.txt");
	cout.rdbuf(fout.rdbuf()); // assign file's streambuf to cout
#ifdef _C_LAN_
	freopen("../output.txt", "w", stdout);
#endif
#endif

	while (cin >> num && num > 0)
	{
		if (num == 1)
		{
			cout << 1 << endl;
			continue;
		}
		if_find = false;
		memset(ans, 0, sizeof(ans));
		ans[0] = 1;
		dfs(1, 1 % num);
	}

	return 0;
}
