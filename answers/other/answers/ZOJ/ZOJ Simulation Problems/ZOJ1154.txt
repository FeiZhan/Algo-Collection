#include<iostream>
#include <string>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	int CaseNum, base;
	long num1, i, num2;
	string digit;
	int p;
	cin>>CaseNum;
	while (CaseNum --)
	{
		while (cin>>base && base != 0)
		{
			cin>>digit;
			num1 = num2 = 0;
			p = 1;
			for (i=0; i<digit.size(); i++)
			{
				num1 *= base;
				num1 += int (digit.at(i) - '0');
				num2 += int (digit.at(i) - '0');
			}
			if (num1 % num2 == 0)
			{
				cout<<"yes"<<endl;
			}
			else cout<<"no"<<endl;
		}
		if (CaseNum > 0)
		{
			cout<<endl;
		}
	}
	//getchar();getchar();
	return 0;
}
