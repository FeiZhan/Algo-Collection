long long Gcd(long long num1, long long num2)
{
	return (num2 == 0) ? num1 : Gcd(num2, num1 % num2);
}
long long Lcm(long long num1, long long num2)
{
	if (num1 < num2)
		swap(num1, num2);
	long long gcd = Gcd(num1, num2);
	if (gcd == 0)
		return 0;
	return num1 * num2 / gcd;
}
