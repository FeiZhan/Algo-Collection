#include<stdio.h>
#include<string.h>
int main()
{
	int length,i,repeat;
	char word[21];

	while(gets(word)!=NULL)
	{
		length=strlen(word);
		repeat=0;
		for(i=0;i<length;i++)
		{
			switch(word[i]-65)
			{
			case 1:case 5:case 15:case 21:
				if(repeat==1) break;
				else {printf("1");repeat=1;break;}
			case 2:case 6:case 9:case 10:case 16:case 18:case 23:case 25:
				if(repeat==2) break;
				else {printf("2");repeat=2;break;}
			case 3:case 19:
				if(repeat==3) break;
				else{printf("3");repeat=3;break;}
			case 11:
				if(repeat==4) break;
				else{printf("4");repeat=4;break;}
			case 12:case 13:
				if(repeat==5) break;
				else{printf("5");repeat=5;break;}
			case 17:
				if(repeat==6) break;
				else{printf("6");repeat=6;break;}
			default: repeat=0;
			}
		}
		printf("\n");
	}
	return 0;
}
