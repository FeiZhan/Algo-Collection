#include <iostream>
#include <string>
//#include <fstream>

using namespace std;
const string MORSE_CODE[30] = {".-", "-...", "-.-.", "-..", ".", "..-.", "--."
	, "....", "..", ".---", "-.-", ".-..", "--", "-."
	, "---", ".--.", "--.-", ".-.", "...", "-", "..-"
	, "...-", ".--", "-..-", "-.--", "--.."
	, "..--", ".-.-", "---.", "----"};
int main()
{
	//ifstream filein("..//data.txt", ios::in);

	int case_num;
	cin>>case_num;
	string input_str, input_code, output_str, code_tmp;
	int code_len[100], code_len2[100], len, code_num;
	char code_chr;
	for (int i = 1; i <= case_num; i ++)
	{
		cin>>input_str;
		len = input_str.size();
		input_code.clear();
		for (int j = 0; j < len; j ++)
		{
			code_chr = input_str.at(j);
			switch (code_chr)
			{
			case '_':
				code_num = 26;
				break;
			case ',':
				code_num = 27;
				break;
			case '.':
				code_num = 28;
				break;
			case '?':
				code_num = 29;
				break;
			default:
				code_num = code_chr - 'A';
			}
			input_code += MORSE_CODE[code_num];
			code_len[j] = MORSE_CODE[code_num].size();
		}
		for (int j = 0; j < len; j ++)
		{
			code_len2[j] = code_len[len - j - 1];
		}
		output_str.clear();
		int search_ptr = 0;
		for (int j = 0; j < len; j ++)
		{
			code_tmp = input_code.substr(search_ptr, code_len2[j]);
			search_ptr += code_len2[j];
			for (int k = 0; k < 31; k++)
			{
				if (code_tmp == MORSE_CODE[k])
				{
					code_num = k;
					break;
				}
			}
			switch (code_num)
			{
			case 26:
				code_chr = '_';
				break;
			case 27:
				code_chr = ',';
				break;
			case 28:
				code_chr = '.';
				break;
			case 29:
				code_chr = '?';
				break;
			default:
				code_chr = code_num + 'A';
			}
			output_str += code_chr;
		}
		cout<<i<<": "<<output_str<<endl;
	}

	//getchar();getchar();
	return 0;
}
