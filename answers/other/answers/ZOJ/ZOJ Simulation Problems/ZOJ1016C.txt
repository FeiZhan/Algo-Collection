#include <stdio.h>
int main()
{
	int i,j,k,ptr,t,n,p[20],w[20],s[40],parenthesis;

	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		scanf("%d",&p[0]);
		for(i=0;i<p[0];i++)
		{
			s[i]=1;
		}
		s[p[0]]=2;
		ptr=p[0];

		for(i=1;i<n;i++)
		{
			scanf("%d",&p[i]);
			for(j=0;j<p[i]-p[i-1];j++)
			{
				s[ptr+j+1]=1;
			}
			ptr+=p[i]-p[i-1]+1;
			s[ptr]=2;
		}

		k=-1;
		for(i=0;i<2*n;i++)
		{
			if(s[i]==2)
			{
				k++;
				w[k]=0;
				parenthesis=0;
				for(j=i;j>=0;j--)
				{
					if(s[j]==1) parenthesis-=1;
					if(s[j]==2)
					{
						parenthesis+=1;
						w[k]+=1;
					}
					if(parenthesis==0) break;
				}
			}
		}
		for(i=0;i<n-1;i++)
		{
			printf("%d ",w[i]);
		}
		printf("%d\n",w[n-1]);
	}
	getchar();
	return 0;
}
