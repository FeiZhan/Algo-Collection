#include <iostream>

using namespace std;

int main(int argc, char *argv[])
{
#ifdef _FILE_DEBUG_
	ifstream fin;
	fin.open("../input.txt");
	cin.rdbuf(fin.rdbuf()); // assign file's streambuf to cin
#ifdef _C_LAN_
	freopen("../input.txt", "r", stdin);
#endif
#endif
#ifdef _FILE_DEBUG_
	ofstream fout;
	fout.open("../output.txt");
	cout.rdbuf(fout.rdbuf()); // assign file's streambuf to cout
#ifdef _C_LAN_
	freopen("../output.txt", "w", stdout);
#endif
#endif

	int case_sum, trip_sum, trip[20], tmp, tmp2;
	double rate[5][5], money;
	cin >> case_sum;
	while (case_sum --)
	{
		for (int i = 0; i < 5; ++ i)
			for (int j = 0; j < 5; ++ j)
			{
				cin >> rate[i][j];
			}
		while (cin >> trip_sum && trip_sum > 0)
		{
			trip[0] = 1;
			for (int i = 1; i <= trip_sum; ++ i)
			{
				cin >> trip[i];
			}
			trip[trip_sum + 1] = 1;
			cin >> money;
			money = int(money * 100 + .5);
			for (int i = 0; i < trip_sum + 1; ++ i)
			{
				money = floor(money * rate[trip[i] - 1][trip[i + 1] - 1] + .5);
			}
			cout << setiosflags(ios::fixed) << setprecision(2)
				<< money / 100.0 << endl;
		}
		if (case_sum)
			cout << endl;
	}

	return 0;
}
