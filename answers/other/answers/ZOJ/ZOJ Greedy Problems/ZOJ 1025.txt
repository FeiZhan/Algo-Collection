//1025Wooden Sticks
#include <iostream>
//#include <fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);

	int case_num;
	cin>>case_num;

	int stick_num, stick[5000][3], temp, count, base_stick;
	while (case_num --)
	{
		cin>>stick_num;
		for (int i = 0; i < stick_num; i ++)
		{
			cin>>stick[i][0]>>stick[i][1];
			stick[i][2] = NULL;
			for (int j = 0; j < i; j ++)
			{
				if (stick[i][0] < stick[j][0]
					|| (stick[i][0] == stick[j][0] && stick[i][1] < stick[j][1]))
				{
					temp = stick[i][0];
					stick[i][0] = stick[j][0];
					stick[j][0] = temp;
					temp = stick[i][1];
					stick[i][1] = stick[j][1];
					stick[j][1] = temp;
				}
			}
		}
		count = 0;
		for (int i = 0; i < stick_num; i ++)
		{
			if (stick[i][2] == NULL)
			{
				count ++;
				base_stick = i;
				for (int j = i + 1; j <stick_num; j ++)
				{
					if (stick[j][2] != NULL)
					{
						continue;
					}
					else if (stick[j][1] >= stick[base_stick][1])
					{
						stick[j][2] = true;
						base_stick = j;
					}
				}
			}
		}
		cout<<count<<endl;
	}

	//system("pause");
	return 0;
}
