#include<iostream>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	int size, mat[100][100], i, j, RowSum[100], ColumnSum[100], RowOdd[100], ColOdd[100]
		, RowOddNum, ColOddNum;
	while (cin>>size && size != 0)
	{
		for (i=0; i<size; i++)
		{
			for (j=0; j<size; j++)
			{
				cin>>mat[i][j];
			}
		}
		RowOddNum = ColOddNum = 0;
		for (i=0; i<size; i++)
		{
			RowSum[i] = ColumnSum[i] = 0;
			for (j=0; j<size; j++)
			{
				RowSum[i] += mat[i][j];
				ColumnSum[i] += mat[j][i];
			}
			if (RowSum[i] % 2 != 0)
			{
				RowOddNum ++;
				RowOdd[RowOddNum - 1] = i + 1;
			}
			if (ColumnSum[i] % 2 != 0)
			{
				ColOddNum ++;
				ColOdd[ColOddNum - 1] = i + 1;
			}
		}
		if (RowOddNum == 0 && ColOddNum == 0)
		{
			cout<<"OK"<<endl;
		}
		else if (RowOddNum == ColOddNum)
		{
			cout<<"Change bit ";
			for (i=0; i<RowOddNum; i++)
			{
				cout<<"("<<RowOdd[i]<<","<<ColOdd[i]<<")";
			}
			cout<<endl;
		}
		else
			cout<<"Corrupt"<<endl;
	}
	//getchar();getchar();
	return 0;
}
