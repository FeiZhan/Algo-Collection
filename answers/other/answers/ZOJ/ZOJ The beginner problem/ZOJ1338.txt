//1338Up and Down Sequences
#include<iostream>
#include<iomanip>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	int num[31], state, neither, up[30], down[30], upnum, downnum, sum, i;
	double upavg, downavg;
	while(cin>>num[0] && num[0] != 0)
	{
		state = neither = upnum = downnum = 0;
		for(i=0; i<30; i++)
		{
			up[i] = down[i] = 0;
		}
		i = 1;
		while(cin>>num[i] && num[i] != 0)
		{
			if(state == 0)
			{
				if(num[i] == num[i-1])
					neither ++;
				else if(num[i] > num[i-1])
				{
					upnum = 0;
					up[upnum] = neither + 1;
					state = 1;
				}
				else if(num[i] < num[i-1])
				{
					downnum = 0;
					down[downnum] = neither + 1;
					state = 2;
				}
			}
			else if(num[i] > num[i-1])
			{
				if(state == 1)
					up[upnum] ++;
				else
				{
					if(up[upnum] != 0)
						upnum ++;
					up[upnum] = 1;
					state = 1;
				}
			}
			else if(num[i] < num[i-1])
			{
				if(state == 2)
					down[downnum] ++;
				else
				{
					if(down[downnum] != 0)
						downnum ++;
					down[downnum] = 1;
					state = 2;
				}
			}
			else if(state == 1 && num[i] == num[i-1])
				up[upnum] ++;
			else if(state == 2 && num[i] == num[i-1])
				down[downnum] ++;
			i++;
		}
		sum = i;
		upavg = downavg = 0;
		for(i=0; i<=upnum; i++)
		{
			upavg += up[i];
		}
		upavg /= double(upnum+1);
		for(i=0; i<=downnum; i++)
		{
			downavg += down[i];
		}
		downavg /= double(downnum+1);
		cout<<"Nr values = "<<sum<<":  "<<setprecision(6)<<setiosflags(ios::fixed)<<upavg<<" "<<downavg<<endl;
	}
	//getchar();
	return 0;
}
