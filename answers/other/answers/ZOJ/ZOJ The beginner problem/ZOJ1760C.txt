#include "stdio.h"
void main()
{
    int a[16],i,j,k,sum;                  /*15个元素+1个0，共16*/
	while(scanf("%d",&a[0]) && a[0]!=-1)
	{
		sum=0;
		for(i=1;i<16;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]==0) break;
		}
		for(j=0;j<i;j++)
			for(k=0;k<i;k++)
			{
				if(a[k]==a[j]*2) sum++;
			}
		printf("%d\n",sum);
	}
}
