#include <stdio.h>
int main()
{
	int line,n,i,j,cell[101],sum;

	scanf("%d",&line);
	while(line--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
			cell[i]=0;
		for(i=1;i<=n;i++)
			for(j=i;j<=n;j+=i)
			{
				cell[j-1]++;
				if(cell[j-1]>=2) cell[j-1]=0;
			}
		sum=0;
		for(i=0;i<n;i++)
		{
			if(cell[i]==1) sum++;
		}
		printf("%d\n",sum);
	}
}
