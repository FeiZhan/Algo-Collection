#include<iostream>
#include<string>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	string classes;
	int mile, sum;
	while(cin>>classes && classes != "#")
	{
		sum = 0;
		while(classes != "0")
		{
			cin>>classes>>mile>>classes;
			if(classes == "F")
			{
				sum += mile * 2;
			}
			else if(classes == "B")
			{
				sum += mile * 1.5;
			}
			else if(classes == "Y" && mile <= 500)
			{
				sum += 500;
			}
			else if(classes == "Y" && mile > 500)
			{
				sum += mile;
			}
			cin>>classes;
		}
		cout<<sum<<endl;
	}
	//getchar();
	return 0;
}
