#include<iostream>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	int n, x, y;
	cin>>n;
	while(n--)
	{
		cin>>x>>y;
		if(x == y && x/2. == double(x/2))
		{
			cout<<2 * x<<endl;
		}
		else if(x == y && x/2. != double(x/2))
		{
			cout<<2 * x - 1<<endl;
		}
		else if(x - 2 >= 0 && x - 2 == y && x/2. == double(x/2))
			cout<<2 * x - 2<<endl;
		else if(x - 2 >= 0 && x - 2 == y && x/2. != double(x/2))
			cout<<2 * x - 3<<endl;
		else cout<<"No Number"<<endl;
	}
	//getchar();getchar();
	return 0;
}
