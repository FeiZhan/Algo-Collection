#include<iostream>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	long long x, y, Recx[2], Recy[2];
	while (cin>>Recx[0]>>Recy[0] && (Recx[0] != 0 || Recy[0] != 0))
	{
		Recx[1] = Recx[0];
		Recy[1] = Recy[0];
		while (cin>>x>>y && (x != 0 || y != 0))
		{
			if (Recx[0] > x)
				Recx[0] = x;
			if (Recx[1] < x)
				Recx[1] = x;
			if (Recy[0] > y)
				Recy[0] = y;
			if (Recy[1] < y)
				Recy[1] = y;
		}
		cout<<Recx[0]<<" "<<Recy[0]<<" "<<Recx[1]<<" "<<Recy[1]<<endl;
	}
	//getchar();getchar();
	return 0;
}
