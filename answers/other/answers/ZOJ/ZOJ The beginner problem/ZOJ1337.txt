#include<iostream>
#include<math.h>
#include<iomanip>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	int n, i, j, k, num[51], factor[200], number, ratio;
	while(cin>>n && n != 0)
	{
		ratio = 0;
		for(i=0; i<n; i++)
		{
			cin>>num[i];
		}
		for(i=0; i<n; i++)
		{
			number = num[i];
			k = 0;
			while(number > 1)
			{
				for(j=2; j<=num[i]; j++)
				{
					if(double(number)/double(j) == double(number/j))
					{
						factor[k] = j;
						k ++;
						number /= j;
						break;
					}
				}
			}
			//if(k == 0)
			//{
			//	factor[0] = number;
			//	k++;
			//}
			for(j=k; j<200; j++)
				factor[j] = 0;
			bool flag;
			for(j=0; j<i; j++)
			{
				flag = false;
				for(k=0; k<200; k++)
				{
					if(factor[k] == 0) break;
					if(double(num[j])/double(factor[k]) == double(num[j]/factor[k]))
					{
						flag = true;
						break;
					}
				}
				if(flag == false)
					ratio ++;
			}
		}
		if(ratio == 0)
			cout<<"No estimate for this data set."<<endl;
		else cout<<setprecision(6)<<setiosflags(ios::fixed)
			<<sqrt(3. * n * (n-1) / ratio)<<endl;
	}
	//getchar();
	return 0;
}
