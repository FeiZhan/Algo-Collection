//2812Quicksum
#include <iostream>
//#include <fstream>
#include <string>
using namespace std;
int main()
{
	//ifstream fin;
	//fin.open("data.txt");
	//cin.rdbuf(fin.rdbuf()); // assign file's streambuf to cin

	int sum, num;
	string str;
	while (getline(cin, str) && str != "#")
	{
		sum = 0;
		for (int i = 0; i < str.size(); i ++)
		{
			num = str.at(i) - 'A' + 1;
			if (num >= 0 && num <= 26)
			{
				sum += (i + 1) * num;
			}
		}
		cout<<sum<<endl;
	}
	//system("pause");
	return 0;
}
