#include<iostream>
#include<math.h>
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	int n, height[100], i = 0, j, sum, move;
	while(cin>>n && n != NULL)
	{
		i++;
		cout<<"Set #"<<i<<endl;
		sum = 0;
		for(j=0; j<n; j++)
		{
			cin>>height[j];
			sum += height[j];
		}
		move = 0;
		for(j=0; j<n; j++)
		{
			move += abs(sum / n - height[j]);
		}
		cout<<"The minimum number of moves is "<<move/2<<"."<<endl<<endl;
	}
	//getchar();
	return 0;
}
