//1051A New Growth Industry
#include<iostream>
#include<fstream>
using namespace std;
//#define cin filein
int main()
{
	int n, day, dnarule[16], i, j, density[20][20], dna[20][20], newdensity[20][20];
	//ifstream filein("..\\data.txt", ios::in);
	cin>>n;
	while(n--)
	{
		cin>>day;
		for(i=0; i<16; i++)
			cin>>dnarule[i];
		for(i=0; i<20; i++)
			for(j=0; j<20; j++)
			{
				cin>>density[i][j];
			}
		while(day--)
		{
			for(i=0; i<20; i++)
				for(j=0; j<20; j++)
				{
					dna[i][j] = density[i][j];
					if(i > 0)
						dna[i][j] += density[i-1][j];
					if(i < 19)
						dna[i][j] += density[i+1][j];
					if(j > 0)
						dna[i][j] += density[i][j-1];
					if(j < 19)
						dna[i][j] += density[i][j+1];
					newdensity[i][j] = density[i][j] + dnarule[dna[i][j]];
					if(newdensity[i][j] < 0)
						newdensity[i][j] = 0;
					if(newdensity[i][j] > 3)
						newdensity[i][j] = 3;
				}
			for(i=0; i<20; i++)
				for(j=0; j<20; j++)
				{
					density[i][j] = newdensity[i][j];
				}
		}
		for(i=0; i<20; i++)
		{
			for(j=0; j<20; j++)
			{
				switch(density[i][j])
				{
				case 0:
					cout<<".";
					break;
				case 1:
					cout<<"!";
					break;
				case 2:
					cout<<"X";
					break;
				case 3:
					cout<<"#";
					break;
				}
			}
			cout<<endl;
		}
		if(n>0) cout<<endl;
		//getchar();
	}
	return 0;
}
