#define _FILE_DEBUG_

//#define _DEBUG_OUTPUT_
#ifdef _FILE_DEBUG_
#include <fstream>
#endif
#include <iostream>
#include <stdio.h>
using namespace std;
#include <string>
#include <memory.h>

int main(int argc, char *argv[])
{
#ifdef _FILE_DEBUG_
	ifstream fin;
	fin.open("../input.txt");
	cin.rdbuf(fin.rdbuf()); // assign file's streambuf to cin
#ifdef _C_LAN_
	freopen("../input.txt", "r", stdin);
#endif
#endif
#ifdef _FILE_DEBUG_
	ofstream fout;
	fout.open("../output.txt");
	cout.rdbuf(fout.rdbuf()); // assign file's streambuf to cout
#ifdef _C_LAN_
	freopen("../output.txt", "w", stdout);
#endif
#endif

	string str;
	int cnt[4];
	while (cin >> str)
	{
		memset(cnt, 0, sizeof(cnt));
		for (string::size_type i = 0; i < str.size(); ++ i)
		{
			switch (str.at(i))
			{
			case 'Z':
				cnt[0] ++;
				str.erase(i, 1);
				i --;
				break;
			case 'O':
				cnt[1] ++;
				str.erase(i, 1);
				i --;
				break;
			case 'J':
				cnt[2] ++;
				str.erase(i, 1);
				i --;
				break;
			case '7':
				cnt[3] ++;
				str.erase(i, 1);
				i --;
				break;
			}
		}
		for (size_t i = 0; i < 4; ++ i)
		{
			for (int j = 0; j < cnt[i]; ++ j)
			{
				switch (i)
				{
				case 0:
					cout << 'Z';
					break;
				case 1:
					cout << 'O';
					break;
				case 2:
					cout << 'J';
					break;
				case 3:
					cout << '7';
				}
			}
		}
		cout << str << endl;
	}

	return 0;
}
