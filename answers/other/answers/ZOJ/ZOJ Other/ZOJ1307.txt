#define _FILE_DEBUG_
//#define _C_LAN_

#ifdef _FILE_DEBUG_
#include <fstream>
#endif
#include <iostream>
#include <stdio.h>
using namespace std;

int main(int argc, char *argv[])
{
#ifdef _FILE_DEBUG_
	ifstream fin;
	fin.open("../input.txt");
	cin.rdbuf(fin.rdbuf());
#ifdef _C_LAN_
	freopen("../input.txt", "r", stdin);
#endif
#endif
#ifdef _FILE_DEBUG_
	ofstream fout;
	fout.open("../output.txt");
	cout.rdbuf(fout.rdbuf());
#ifdef _C_LAN_
	freopen("../output.txt", "w", stdout);
#endif
#endif

	int num[6], sum;
	bool is_end;
	while (true)
	{
		is_end = true;
		for (int i = 0; i < 6; ++ i)
		{
			cin >> num[i];
			if (num[i] > 0)
			{
				is_end = false;
			}
		}
		if (is_end == true)
		{
			break;
		}
		sum = num[5];

		sum += num[4];
		num[0] -= num[4] * 11;

		sum += num[3];
		if (num[1] >= num[3] * 5)	//
		{
			num[1] -= num[3] * 5;
		}
		else
		{
			num[0] -= 4 * (num[3] * 5 - num[1]);
			num[1] = 0;
		}

		sum += num[2] / 4;
		num[2] %= 4;
		switch (num[2])
		{
		case 3:
			if (num[1] > 0)
			{
				-- num[1];
				num[0] -= 5;
			}
			else
			{
				num[0] -= 9;
			}
			++ sum;
			break;
		case 2:
			if (num[1] >= 3)
			{
				num[1] -= 3;
				num[0] -= 6;
			}
			else
			{
				num[0] -= 6 + (3 - num[1]) * 4;
				num[1] = 0;
			}
			++ sum;
			break;
		case 1:
			if (num[1] >= 5)
			{
				num[1] -= 5;
				num[0] -= 7;
			}
			else
			{
				num[0] -= 7 + (5 - num[1]) * 4;
				num[1] = 0;
			}
			++ sum;
			break;
		}

		sum += num[1] / 9;
		num[1] %= 9;
		if (num[1] > 0)
		{
			++ sum;
			num[0] -= (9 - num[1]) * 4;
		}

		if (num[0] > 0)
		{
			sum += num[0] / 36;
			num[0] %= 36;
			if (num[0] > 0)
			{
				++ sum;
			}
		}
		cout << sum << endl;
	}

	return 0;
}
