#include <iostream>

#ifdef DEBUG
#include <fstream>
#endif

using namespace std;
int main()
{
#ifdef DEBUG
	ifstream fin;
	fin.open("data.txt");
	cin.rdbuf(fin.rdbuf()); // assign file's streambuf to cin
#endif

	int case_sum, turkey_sum, digit[5], total_price;
	bool is_find;
	cin >> case_sum;
	while (case_sum --)
	{
		cin >> turkey_sum >> digit[1] >> digit[2] >> digit[3];
		is_find = false;
		for (int i = 9; i > 0; i --)
		{
			for (int j = 9; j >= 0; j --)
			{
				total_price = i * 10000 + digit[1] * 1000 + digit[2] * 100 + digit[3] * 10
					+ j;
				if (total_price % turkey_sum == 0)
				{
					is_find = true;
					cout << i << " " << j << " " << total_price / turkey_sum << endl;
					break;
				}
			}
			if (is_find == true)
			{
				break;
			}
		}
		if (is_find == false)
		{
			cout << 0 << endl;
		}
	}

#ifdef DEBUG
	system("pause");
#endif
	return 0;
}
