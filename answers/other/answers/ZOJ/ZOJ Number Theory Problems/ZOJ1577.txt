#include <iostream>
#include <stdio.h>
using namespace std;
#include <cmath>

int gcd(int m, int n)
{
    return n == 0 ? m : gcd(n,m % n);
}
int main()
{
#ifdef _FILE_DEBUG_
	ifstream fin;
	fin.open("../input.txt");
	cin.rdbuf(fin.rdbuf()); // assign file's streambuf to cin
#ifdef _C_LAN_
	freopen("../input.txt", "r", stdin);
#endif
#endif
#ifdef _FILE_DEBUG_
	ofstream fout;
	fout.open("../output.txt");
	cout.rdbuf(fout.rdbuf()); // assign file's streambuf to cout
#ifdef _C_LAN_
	freopen("../output.txt", "w", stdout);
#endif
#endif

	int x, y, p_mul_q, factor, sum, i;
	while (cin >> x >> y)
	{
		if (y % x != 0)
		{
			cout << 0 << endl;
			continue;
		}
		if (x == y)
		{
			cout << 1 << endl;
			continue;
		}
		sum = 0;
		p_mul_q = x * y;
		for (i = 1; i < sqrt(double(p_mul_q)); i ++)
		{
			if (p_mul_q % i == 0)
			{
				factor = p_mul_q / i;
				if (gcd(factor, i) == x)
				{
					sum ++;
				}
			}
		}
		sum *= 2;
		if (factor == i)
			sum --;
		cout << sum << endl;
	}
	return 0;
}
