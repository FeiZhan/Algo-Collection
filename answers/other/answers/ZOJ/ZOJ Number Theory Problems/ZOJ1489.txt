//14892^x mod n = 1
#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
	int num, base_num, power_num;
	while(cin>>num && cin.eof() == 0)
	{
		if (num % 2 == 0 || num == 1)
		{
			cout<<"2^? mod "<<num<<" = 1"<<endl;
		}
		else
		{
			base_num = 2;
			power_num = 1;
			while (base_num != 1)
			{
				power_num ++;
				base_num *= 2;
				base_num %= num;
			}
			printf("2^%d mod %d = 1\n", power_num, num);
		}
	}
	//system("pause");
	return 0;
}
