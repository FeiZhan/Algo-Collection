#ifdef _FILE_IN_
#include <fstream>
#endif

#include <iostream>
#include <stdio.h>
#include <string>
#include <vector>
using namespace std;

int main()
{
#ifdef _FILE_IN_
	ifstream fin;
	fin.open("input.txt");
	cin.rdbuf(fin.rdbuf()); // assign file's streambuf to cin
#ifdef _C_LAN_
	freopen("input.txt", "r", stdin);
#endif
#endif
#ifdef _FILE_OUT_
	ofstream fout;
	fout.open("output.txt");
	cout.rdbuf(fout.rdbuf()); // assign file's streambuf to cout
#ifdef _C_LAN_
	freopen("output.txt", "w", stdout);
#endif
#endif

	string str_in, page_now = "http://www.acm.org/";
	vector<string> back, forw;

	do 
	{
		getline(cin, str_in);
		switch (str_in.at(0))
		{
		case 'V':
			if (page_now.size() != NULL)
			{
				back.push_back(page_now);
			}
			page_now = str_in.substr(6);
			cout << page_now << endl;
			forw.clear();
			break;
		case 'B':
			if (back.size() == 0)
			{
				cout << "Ignored" << endl;
			}
			else
			{
				forw.push_back(page_now);
				page_now = back.back();
				back.erase(back.end() - 1);
				cout << page_now << endl;
			}
			break;
		case 'F':
			if (forw.size() == 0)
			{
				cout << "Ignored" << endl;
			}
			else
			{
				back.push_back(page_now);
				page_now = forw.back();
				forw.erase(forw.end() - 1);
				cout << page_now << endl;
			}
			break;
		case 'Q':
		default:
			break;
		}
	} while (str_in.compare("QUIT"));

#ifdef _FILE_IN_
#ifndef _FILE_OUT_
	system("pause");
#endif
#endif
	return 0;
}
