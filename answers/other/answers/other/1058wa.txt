#include<iostream>
using namespace std;
int main()
{
	int n, i, j, tripnum, trip[10], ctripn;
	double rate[5][5], initmoney, cmoney;
	cin>>n;
	while(n--)
	{
		for(i=0; i<5; i++)
			for(j=0; j<5; j++)
			{
				cin>>rate[i][j];
			}
		cin>>tripnum;
		while(tripnum != 0)
		{
			for(i=0; i<tripnum; i++)
			{
				cin>>trip[i];
			}
			cin>>initmoney;
			ctripn = 0;
			cmoney = initmoney;
			for(i=0; i<tripnum; i++)
			{
				cmoney *= rate[ctripn][trip[i]-1];
				cmoney = (int)(cmoney * 100. + 0.5) / 100.;
				ctripn = trip[i]-1;
			}
			cmoney *= rate[ctripn][0];
			cmoney = (int)(cmoney * 100. + 0.5) / 100.;
			if(cmoney < 1) cout<<"0";
			cout<<cmoney<<endl;
			getchar();
			cin>>tripnum;
		}
	}
	return 0;
}
