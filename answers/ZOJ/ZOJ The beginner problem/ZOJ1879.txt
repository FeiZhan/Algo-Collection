//1879Jolly Jumpers
#include<iostream>
#include <math.h>
#define MAX 3000
//#include<fstream>
//#define cin filein
using namespace std;
int main()
{
	//ifstream filein("..//data.txt", ios::in);
	int num[MAX], diff[MAX], i, sequence[MAX], MaxNum;
	bool ExitFlag;
	while (cin>>MaxNum != NULL)
	{
		cin>>num[0];
		sequence[0] = 0;
		for (i=1; i<MaxNum; i++)
		{
			cin>>num[i];
			sequence[i] = 0;
			diff[i - 1] = abs(num[i] - num[i - 1]);
		}
		for (i=0; i<MaxNum - 1; i++)
		{
			sequence[diff[i] - 1] ++;
		}
		if (MaxNum == 1)
		{
			cout<<"Jolly"<<endl;
		}
		else
		{
			ExitFlag = false;
			for (i=0; i<MaxNum - 1; i++)
			{
				if (sequence[i] == 0)
				{
					ExitFlag = true;
					cout<<"Not jolly"<<endl;
					break;
				}
			}
			if (ExitFlag == false)
			{
				cout<<"Jolly"<<endl;
			}
		}
	}
	//getchar();getchar();
	return 0;
}
